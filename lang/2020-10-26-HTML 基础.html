HTML 基础
集成开发工具：zend stdio 和 netbean 。前端notepad++ 和editplus 元素：标签和标签内容被抱起的，叫元素。 HTML实体：http://www.w3school.com.cn/html/html_entities.asp 

<!DOCTYPE html>
<html>
<head>
  <title>标题</title>
  <meta charset="utf-8">
</head>
<body>
<div> <!-- <div></div>标签是布局 -->
  <h3>这是标题</h3> <!-- h1到h5的数字 -->
  <hr>  <!-- 换行 -->
  <!-- -------------------------------------------------------------------------------------------------------- -->
  <marquee direction="left">文字滚动</marquee>    <!-- 往左边滚动 -->
  <font color="red" size="15">字体颜色大小演示</font><br>
  <i>字体样式</i><br> <!-- 斜体 -->
  <u>字体样式</u><br> <!-- 下划线 -->
  <b>字体样式</b><br> <!-- 黑体 -->
  <s>字体样式</s> <!-- 中间下划线 -->
  <span>我是span标签!</span>  <!-- span>在行内定义一个区域 -->
  <!-- -------------------------------------------------------------------------------------------------------- -->
  36<sup>o</sup>C   <!-- 上标   -->
  H<sub>2</sub>O    <!-- 下标 -->
    <ul>      <!-- 无序列表 -->
      <li>无序列表1</li>
      <li>无序列表2</li>
    </ul>
    <ol>      <!-- 有序列表 -->
      <li>有序列表1</li>
      <li>有序列表1</li>
    </ol>
  <!-- -------------------------------------------------------------------------------------------------------- -->
  <img src="image/asa.png" alt="图片的描述" width="200px" height="160px" /><br>  <!-- 图片标签 -->
  <embed src="image/videoplayback.mp4" width="480" height="360"></embed><br>  <!-- 多媒体标签，视频，音频，flash -->
  
  <img src="image/asa.png"  usemap="#testmap" alt="test" />
    <map name="testmap" id="testmap">
        <area shape="circle" coords="180,139,14" href ="test1.html" alt="test1" />
        <area shape="circle" coords="129,161,10" href ="test2.html" alt="test2" />
        <area shape="rect" coords="0,0,110,260" href ="test3.html" alt="test3" />
    </map>
</div>
</body>
</html>
