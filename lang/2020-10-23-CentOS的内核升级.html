CentOS 的内核升级.html
1. 连接并同步CentOS自带yum源，更新内核版本。此方法适用于更新内核补丁 。 

uname -r    #查看内核版
yum list kernel
yum update -y kernel
第二，大版本升级载入elrepo源，搜索内核更新资源，并进行更新操作

rpm --import https://www.elrepo.org/RPM-GPG-KEY-elrepo.org    # 载入公钥
rpm -Uvh https://www.elrepo.org/elrepo-release-7.0-3.el7.elrepo.noarch.rpm    # 安装ELRepo
yum --disablerepo=\* --enablerepo=elrepo-kernel repolist  # 载入elrepo-kernel元数据
yum --disablerepo=\* --enablerepo=elrepo-kernel list kernel*  # 查看可用的rpm包
yum --disablerepo=\* --enablerepo=elrepo-kernel install -y kernel-ml.x86_64   # 安装最新版本的kernel

###最后一步，需要将内核工具包一并升级
yum remove kernel-tools-libs.x86_64 kernel-tools.x86_64   # 删除旧版本工具包
yum --disablerepo=\* --enablerepo=elrepo-kernel install -y kernel-ml-tools.x86_64   # 安装新版本工具包
开启BBR

vim /etc/sysctl.conf

net.core.default_qdisc = fq
net.ipv4.tcp_congestion_control = bbr
sysctl -p ###加载系统参数（正常情况下会输出我们之前加入的内容）

sysctl net.ipv4.tcp_available_congestion_control        ##若输入
net.ipv4.tcp_available_congestion_control = bbr cubic reno  ##返回这个BBR开启

lsmod | grep bbr  ##若输入
##返回这个BBR开启
tcp_bbr                16384  1
tcp_bbr                20480  0
